Select * from Designation 
	where	(@Id = 0 or Id = @Id) and
			(@Name = '' or Name Like '%'+LTRIM(RTRIM(@Name))+'%') and
			(@Type = 0 or Type = @Type) and
			IsActive = 1 and
			IsDeleted = 0
			Order By Type, DisplayOrder
