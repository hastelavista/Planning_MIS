Begin
	Update Employee Set
		Name = @Name,
		Address = @Address,
		MobileNo = @MobileNo,
		DepartmentId = @DepartmentId,
		DesignationId = @DesignationId,
		IsTechnicalPerson = @IsTechnicalPerson,
		IsWardSecratery = @IsWardSecratery,

		ModifiedBy = @CurrentUser
	Where Id = @Id

		Values(@Name, @Address, @MobileNo, @DepartmentId, @DesignationId, @IsTechnicalPerson, @IsWardSecratery, @IsAccountant, GETDATE(), @CurrentUser, Null, Null, 0)
