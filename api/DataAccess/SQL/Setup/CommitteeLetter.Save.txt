If @Id > 0
	Begin
		Update CommitteeLetter Set
			Date = @Date,
			Miti = @Miti,
			CommitteeId = @CommitteeId,
			LetterTypeId = @LetterTypeId,
			LetterFormatId = @LetterFormatId,
			Subject = @Subject,
			LetterBody = @LetterBody,
			ModifiedDate = GETDATE(),
			ModifiedBy = @CurrentUser
		Where Id = @Id
	End
Else
	Begin
		Insert into CommitteeLetter(Date, Miti, CommitteeId, LetterTypeId, LetterFormatId, Subject, LetterBody, CreatedDate, CreatedBy, IsDeleted)
		Values(@Date, @Miti, @CommitteeId, @LetterTypeId, @LetterFormatId, @Subject, @LetterBody, GETDATE(), @CurrentUser, 0)
	End
